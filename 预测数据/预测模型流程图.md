# 足球预测模型完整架构流程图

```mermaid
flowchart TB
    %% ===== 入口 =====
    USER(["用户输入：主队 vs 客队 + 模式关键词"])
    USER --> ROUTER{Step 0<br>模式路由}
    
    %% ===== 竞彩模式 =====
    ROUTER -->|"关键词：<br>竞彩/体彩/串关/不指定"| JC["🟢 竞彩模式"]
    JC --> FETCH_JC["数据获取 fetch_match_data.py<br>--mode jingcai<br>中国竞彩网 API"]
    FETCH_JC --> JC_DATA["竞彩数据<br>• 胜平负赔率<br>• 让球胜平负<br>• 单关标记"]
    JC_DATA --> PIPE_A
    JC_DATA --> PIPE_B
    JC_DATA --> SKIP_C["❌ 跳过管线C<br>（角球/罚牌<br>竞彩不可购买）"]
    SKIP_C --> REPORT_JC["输出竞彩报告<br>4维：胜平负+让球+大小球+半全场"]

    %% ===== 亚盘模式 =====
    ROUTER -->|"关键词：<br>亚盘/欧赔/Bet365/全维度"| YP["🔵 亚盘模式"]
    YP --> FETCH_YP["数据获取 fetch_match_data.py<br>--mode yapan<br>odds-api.io"]
    FETCH_YP --> YP_DATA["亚盘数据<br>• Bet365欧赔/亚盘/大小/角球/罚牌/半全场<br>• 1xbet多线亚盘/大小/角球<br>• 替代盘口/罚牌让球/半场数据"]
    YP_DATA --> PIPE_A
    YP_DATA --> PIPE_B
    YP_DATA --> PIPE_C
    PIPE_C --> PIPELINE_D

    %% ===== 三管线独立分析 =====
    subgraph PIPELINES ["三管线独立分析 + 管线D"]
        PIPE_A["管线A：胜平负/让球"]
        PIPE_B["管线B：大小球"]
        PIPE_C["管线C：角球/罚牌"]
        PIPE_D["管线D：亚盘双阶段"]
    end

    %% ===== 管线A详细 =====
    subgraph PA ["管线A：胜平负/让球"]
        A1["赔率倒推概率<br>欧赔→隐含概率"]
        A2["赢盘率对比<br>主客队赢盘率差"]
        A3["盘口深度评估<br>浅盘/深盘判断"]
        A4["水位变化趋势<br>初盘vs现盘水位"]
        A5["单关/非单关判定<br>诱盘风险检查"]
        A6["半全场方向<br>结合半场数据推演"]
        A1 --> A_CHECK
        A2 --> A_CHECK
    end

    %% ===== 管线B详细 =====
    subgraph PB ["管线B：大小球"]
        B1["战术风格对阵<br>高位逼抢/传控/防反"]
        B2["双向进球评估<br>主X球+客X球=总区间"]
        B2A["半场比分推演 🆕v6.2<br>必须赢队半场落后→×1.5-2.0<br>半场0-0但必须赢→×1.2-1.5"]
        B3["领先后退速评估<br>保平即出线？已领先？"]
        B4["生死战防守提升<br>弱队死守失球↓30-50%"]
        B5["大小球错误自查<br>6种常见错误模式<br>+ 默契球检查 🆕v6.2"]
        B1 --> B2
        B2 --> B2A
        B2A --> B3
        B3 --> B4
        B4 --> B5
        B5 --> B_CHECK
    end

    %% ===== 管线C详细 =====
    subgraph PC ["管线C：角球/罚牌"]
        C1["刷角能力评分<br>战术风格×历史数据"]
        C2["角球基准值<br>比赛类型→预期角球数"]
        C3["角球盘口验证<br>1xbet 25条角球线"]
        C4["罚牌基准计算<br>纪律数据+战意+裁判"]
        C5["罚牌盘口验证<br>Bet365 Bookings Totals"]
        C1 --> C_CHECK
        C4 --> C_CHECK
    end

    %% ===== 管线D详细 =====
    subgraph PD ["管线D：亚盘双阶段"]
        D1["D1 初盘预测（赛前）<br>标准五维分析"]
        D2["D2 滚球预测（赛中）<br>按比赛状态推演"]
        D2 --> D2A["比赛状态分类<br>0-0胶着→首球→半场→<br>体能拐点→尾段"]
        D2 --> D2B["比分情景推演<br>8种比分→最可能走势"]
        D2 --> D2C["角球滚球策略<br>攻防节奏→追大/小角"]
        D2 --> D2D["罚牌滚球策略<br>情绪+时间→红牌预警"]
    end

    %% ===== 矛盾检查 =====
    A_CHECK["管线A 矛盾信号检查"]
    B_CHECK["管线B 矛盾信号检查"]
    C_CHECK["管线C 矛盾信号检查"]

    A_CHECK -->|"矛盾已处理"| A_VERDICT["管线A结论<br>方向+信心+证据分"]
    A_CHECK -->|"矛盾未解决"| A_DOWNGRADE["信心降级"]
    A_DOWNGRADE --> A_VERDICT

    B_CHECK -->|"已处理"| B_VERDICT["管线B结论<br>方向+信心+区间"]
    B_CHECK -->|"未解决"| B_DOWNGRADE["信心降级"]
    B_DOWNGRADE --> B_VERDICT

    C_CHECK -->|"已处理"| C_VERDICT["管线C结论<br>角球+罚牌方向"]
    C_CHECK -->|"未解决"| C_DOWNGRADE["信心降级"]
    C_DOWNGRADE --> C_VERDICT

    %% ===== 汇总 =====
    A_VERDICT --> SUMMARY["Step 7.1 多方向整合"]
    B_VERDICT --> SUMMARY
    C_VERDICT --> SUMMARY
    D1 --> SUMMARY
    D2 --> SUMMARY

    SUMMARY --> REPORT

    %% ===== 报告生成 =====
    subgraph OUTPUT ["报告输出"]
        REPORT["Step 7.2 生成预测报告"]
        REPORT --> SAVE1[("保存<br>predictions.json<br>含全部方向/信心/证据")]
        REPORT --> SAVE2[("保存<br>{主队}vs{客队}_预测.md<br>独立分析报告")]
        REPORT --> DISPLAY["输出到聊天<br>按模式显示正确维度"]
    end

    %% ===== 赛后 =====
    SAVE1 --> POST["赛后结果录入"]
    POST --> DEVIATION["偏差分析<br>实际vs预测对比"]
    DEVIATION --> ROOT_CAUSE["根因分析<br>填写primary/secondary/tertiary"]
    ROOT_CAUSE --> UPDATE_STATS["更新aggregate_stats<br>准确率统计+趋势"]
```
